Conditional statements - extremely important

foo <- rep(NA, 100)
for (i in 1:100) {
    if (i%%2 == 0) {
        foo[i] <- sin(i)
    } else {
        foo[i] <- 3
    }
}

plot(foo, type = "b", col = c("green", "red"), pch = c(8, 20))

These all be can be used in 'functions'

funny.plot <- function(iterations) {
    foo <- rep(NA, iterations)
    for (i in 1:iterations) {
        if (i%%2 == 0) {
            foo[i] <- sin(i)
        } else {
            foo[i] <- 3
        }
    }
    plot(foo, type = "b", col = c("green", "red"), pch = c(8, 20))
}
